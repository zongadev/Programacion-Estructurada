//01B-entrada-salida-variable-tipo
#include <stdio.h> /** Librería: https://es.wikipedia.org/wiki/Stdio.h */
#include <stdlib.h> /** Librería: https://es.wikipedia.org/wiki/Stdlib.h */


int main()
{
    // definición SIN inicialización
    int num; /** definición variable tipo entera*/
    float real; /** definición variable tipo real "float"*/
    double real2; /** definición variable tipo real "double"*/
    // definición CON inicialización
    int diasLaborables = 0; /** definición variable tipo entero "int"*/
    char caracter =' '; /** definición variable tipo caracter "char"*/
    
    char msj[10] = "hola"; /** arreglo de caracteres ...AMPLIAREMOS LA INFO*/
    
    //asignación
    num = 10;
    real = 3.1418;
    real2 = 1234.231418;
    diasLaborables = 22;
    caracter = 'x';
    
    printf("\n---------------------------------------\n");
    printf("Cadena: %s, caracter: %c, entero: %d, flotante: %4.2lf ",msj,
         caracter, num,real2);
    printf("\n---------------------------------------\n");
    
    //INGRESAR: Utilizamos scanf para numeros y getchar para caracteres.
    printf("\n (1)Ingresar valor entero: ");
    scanf("%d",&num); //ingresar (ponemos %d cuando la variable es int)
    //printf("entero: %d", num);
    
    printf("\n (2)Ingresar valor Real(float): ");
    scanf("%f",&real); // Ingresar: (ponemos %f cuando variable es float)
    
    
    printf("\n (3)Ingresar valor Real(double): ");
    scanf("%lf",&real2); // Ingresar: (ponemos %lf cuando variable es double)
    

    //LIMPIEZA DE BUfFER DE ENTRADA. (VER NOTA_01 ABAJO).
    //fflush(stdin); // Sólo windows
    while((caracter = getchar()) != '\n' && caracter != EOF); // muliplataforma
    printf("\n (4)Ingresar valor Caracter: ");
    caracter = getchar();
    
    /**
    La carga de una cadena de caracteres queda pendiente
    para cuando sepamos ciclos
    */    
    
    printf("\n---------------------------------------\n");
    printf("caracter: %c, entero: %d, flotante: %4.2f, double: %6.2lf",caracter,
         num,real, real2);
    printf("\n---------------------------------------\n");
    
    
    return 0;
}
